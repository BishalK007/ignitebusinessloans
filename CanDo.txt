const handleLoaderDone = async () => {
    const MIN_LOADER_TIME = 3000;
    const start = Date.now();
    await submitAnswers();
    const elapsed = Date.now() - start;
    if (elapsed < MIN_LOADER_TIME) {
      setTimeout(() => {
        setShowLoader(false);
        setShowThankYou(true);
      }, MIN_LOADER_TIME - elapsed);
    } else {
      setShowLoader(false);
      setShowThankYou(true);
    }
};
